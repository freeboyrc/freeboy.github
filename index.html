crosscompile gdb

View on GitHub Download .zip Download .tar.gz

1.compile gdb

cd gdb-7.9 ./configure --target=arm-none-linux-gnueabi --disable-tui --prefix=/opt/gdb-7.9 make make install

2.compile gdbserver

cd gdb-7.9/gdb/gdbserver ./configure --target=arm-none-linux-gnueabi --host=arm-none-linux-gnueabi make CC=arm-none-linux-gnueabi-gcc
